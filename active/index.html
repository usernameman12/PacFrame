How to use


<!-- Include the scripts -->
<script src="https://pacframe.pages.dev/active/js/pacman.js"></script>
<script src="https://pacframe.pages.dev/active/js/pacframe.js"></script>

<!-- Use PacFrame -->
<script>
  // Basic usage
  pacframe.embed("https://wikipedia.org", "container-id");
  
  // With options
  pacframe.embed("https://wikipedia.org", "container-id", {
    width: "100%",
    height: "500px",
    showAnimation: true,
    loadingText: "Chomping bytes..."
  });
</script>
